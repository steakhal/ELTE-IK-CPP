% param #1 x coordinate position

% param #3 height

	\draw [thick, black] (#1, #2) -- (#1 + \stackNodeWidth/10, #2);
	\draw [thick, black] (#1, #2) -- (#1, #3);
	\draw [thick, black] (#1 + \stackNodeWidth/10, #2) -- (#1 + \stackNodeWidth/10, #3);
